# 별 모양 만들기

```
// 별이 아래와 같이 2배씩 늘어나며 출력되는 solution 을 완성하세요

fun main() {
    solution(3)
//     *
//     **
//    ****

    solution(5)
//    *
//    **
//    ****
//    ********
//    ****************
}

fun solution(star : Int) {

    var cnt = 1

    for (i in 1..star) {
        for (j in 1..cnt) {
            print("*")
        }
        println()
        cnt *= 2
    }

}
```
